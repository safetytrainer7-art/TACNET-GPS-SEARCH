// 1. Define the different map styles (Layers) const streetLayer = L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap' }); const satelliteLayer = L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', { attribution: 'Tiles © Esri' }); // 2. Initialize the map with the Street layer active by default const map = L.map('map', { center: [37.5407, -77.4360], zoom: 15, layers: [streetLayer] // Sets the default view }); // 3. Create a group for the layers const baseMaps = { "Streets": streetLayer, "Satellite": satelliteLayer }; // 4. Add the Layer Control button to the map L.control.layers(baseMaps).addTo(map);
