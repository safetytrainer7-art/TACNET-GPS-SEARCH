// 1. Define the layers const streetLayer = L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap' }); const satelliteLayer = L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', { attribution: 'Tiles © Esri' }); // 2. Initialize the map with one default layer const map = L.map('map', { center: [37.5407, -77.4360], zoom: 15, layers: [streetLayer] }); // 3. Define the menu options const baseMaps = { "Streets": streetLayer, "Satellite": satelliteLayer }; // 4. Add the toggle control (this creates the button in the top-right) L.control.layers(baseMaps).addTo(map);
